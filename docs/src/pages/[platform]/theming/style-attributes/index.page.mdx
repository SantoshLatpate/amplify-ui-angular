These style attributes are available on amplify-ui primitive components and can be used to modify the corresponding styles. This allows users to quickly make style adjustments without having to go through css or add a theme object.

### Background Styles

<ExampleCode>

```tsx
import { Card, useTheme } from '@aws-amplify/ui-react';

//use any CSS value
<Card backgroundColor="red">Background Styling Example</Card>;

//use a design token from the theme object
const { tokens } = useTheme();
<Card backgroundColor={tokens.colors.brand.primary[40]}>
  Background Styling Example
</Card>;

//use a design token name
<Card backgroundColor="brand.primary.40">Background Styling Example</Card>;
```

</ExampleCode>

### Border Styles

```tsx
import { Button, useTheme } from '@aws-amplify/ui-react';

//use any CSS value
<Button borderRadius="0.5rem">CSS Style</Button>

//using a design token
const { tokens } = useTheme();
<Button borderRadius={tokens.radii.medium}>Token Styling</Button>

//use a design token name
<Button borderRadius="medium">Token Name</Button>
```

### Flex

```tsx
import { Flex, useTheme } from '@aws-amplify/ui-react';

//use any CSS value
<Flex direction="row-reverse" wrap="wrap">
  <Button order="3">Button 1</Button>
  <Button order="1">Button 2</Button>
  <Button order="2">Button 3</Button>
</Flex>;

//using design tokens
<Flex direction="row-reverse" backgroundColor={tokens.colors.neutral[20]}>
  <Button order="3" color={tokens.colors.secondary[80]}>
    Button 1
  </Button>
  <Button order="1" color={tokens.colors.secondary[80]}>
    Button 2
  </Button>
  <Button order="2" color={tokens.colors.secondary[80]}>
    Button 3
  </Button>
</Flex>;

//using token names
<Flex direction="row-reverse" backgroundColor="colors.neutral.20">
  <Button order="3" color="secondary.80">
    Button 1
  </Button>
  <Button order="1" color="secondary.80">
    Button 2
  </Button>
  <Button order="2" color="secondary.80">
    Button 3
  </Button>
</Flex>;
```

### Grid

```tsx
import { Grid, View, useTheme } from '@aws-amplify/ui-react';

//use any CSS value
<Grid templateColumns="1fr 1fr" templateRows="10rem 10rem">
  <View backgroundColor="blue"></View>
  <View backgroundColor="blue"></View>
  <View backgroundColor="blue"></View>
  <View backgroundColor="blue"></View>
</Grid>;

//use design tokens
const { tokens } = useTheme();
<Grid
  templateColumns="1fr 1fr"
  templateRows="10rem 10rem"
  gap={tokens.space.small}
>
  <View backgroundColor={tokens.colors.blue[10]}></View>
  <View backgroundColor={tokens.colors.blue[20]}></View>
  <View backgroundColor={tokens.colors.blue[40]}></View>
  <View backgroundColor={tokens.colors.blue[60]}></View>
</Grid>;

//use token names
<Grid templateColumns="1fr 1fr" templateRows="10rem 10rem" gap="space.small">
  <View backgroundColor="blue.10"></View>
  <View backgroundColor="blue.10"></View>
  <View backgroundColor="blue.10"></View>
  <View backgroundColor="blue.10"></View>
</Grid>;
```
